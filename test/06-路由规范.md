## 路由跳转规范

### API 跳转

```typescript
// 跳转到指定页面
function navigateTo(url: string, params?: Record<string, string>) {
  let path = url
  if (params) {
    const query = Object.entries(params)
      .map(([key, value]) => `${key}=${encodeURIComponent(value)}`)
      .join('&')
    path += `?${query}`
  }
  uni.navigateTo({ url: path })
}

// 跳转到 TabBar 页面
function switchTab(url: string) {
  uni.switchTab({ url })
}

// 返回上一页
function navigateBack(delta = 1) {
  uni.navigateBack({ delta })
}

// 重定向
function redirectTo(url: string) {
  uni.redirectTo({ url })
}

// 重新加载
function reLaunch(url: string) {
  uni.reLaunch({ url })
}
```

### 路由传参

// 传递参数

## 路由守卫

### 全局前置守卫

```typescript
// main.ts
import type { UniAppRouteNext, UniAppRoute } from '@/types'

// 路由白名单（无需登录）
const whiteList = [
  '/pages/index/index',
  '/pages/login/index'
]

// 全局前置守卫
uni.addInterceptor('navigateTo', {
  invoke(params: { url: string }) {
    const token = uni.getStorageSync('token')
    const url = params.url.split('?')[0]

    // 未登录且不在白名单
    if (!token && !whiteList.includes(url)) {
      uni.showToast({ title: '请先登录', icon: 'none' })
      uni.navigateTo({ url: '/pages/login/index' })
      return false
    }
    return true
  }
})
```

## 路由类型定义

```typescript
// types/route.ts
export interface RouteParams {
  [key: string]: string
}

export interface NavigateOptions {
  url: string
  params?: RouteParams
  success?: () => void
  fail?: () => void
}

export type RouteMethod = 'navigateTo' | 'switchTab' | 'redirectTo' | 'reLaunch'
```

## 页面栈管理

```typescript
// utils/route.ts
export const router = {
  // 跳转到页面
  to(url: string, params?: Record<string, string>) {
    let path = url
    if (params) {
      const query = Object.entries(params)
        .map(([k, v]) => `${k}=${encodeURIComponent(v)}`)
        .join('&')
      path += `?${query}`
    }
    return uni.navigateTo({ url: path })
  },

  // 跳转到 TabBar
  toTab(url: string) {
    return uni.switchTab({ url })
  },

  // 返回
  back(delta = 1) {
    return uni.navigateBack({ delta })
  },

  // 替换当前页面
  replace(url: string) {
    return uni.redirectTo({ url })
  },

  // 重新加载
  relaunch(url: string) {
    return uni.reLaunch({ url })
  }
}
```

## 禁止事项

1. **禁止硬编码页面路径**（使用常量管理）
2. **禁止不处理路由失败情况**
3. **禁止循环跳转**（如 A→B→A）
4. **页面路径必须与 pages.json 一致**
